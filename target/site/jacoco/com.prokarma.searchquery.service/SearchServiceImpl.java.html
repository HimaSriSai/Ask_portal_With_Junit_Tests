package com.prokarma.searchquery.service;

 *

	/**
	 *

	/**
	 * 
	 */

	private transient ISearchDao searchDao;

	 * 
	 * @see com.prokarma.search_query.service.ISearchService#getcategory()

	public List<String> getcategory() {

	}

}
